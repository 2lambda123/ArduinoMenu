133 void serial_send(uint8_t data){
134  /*
135  Synchronously send one byte of serial data on the pin specified by
136  SERIAL_PIN. Data is sent at 600 baud
137  */
138  uint8_t mask = 1;
139  //start bit
140  PORTB &= ~_BV(SERIAL_TX_PIN);
141  _delay_loop_3(416);
142  do{
143  if (data & mask){
144  PORTB |= _BV(SERIAL_TX_PIN);
145  }
146  else{
147  PORTB &= ~_BV(SERIAL_TX_PIN);
148  }
149  mask = mask << 1;
150  _delay_loop_3(416);
151  } while(mask);
152  //stop bit
153  PORTB |= _BV(SERIAL_TX_PIN);
154  _delay_loop_3(416);
155 }
